## boj 2309번 문제
https://www.acmicpc.net/problem/2309

### 1. 아홉 난쟁이의 키를 입력받아 리스트에 저장 (숫자로 변환)
nine_dwarfs = []
for _ in range(9):
    nine_dwarfs.append(int(input()))

### 2. 전체 키의 합 구하기
total_sum = sum(nine_dwarfs)

### 3. 가짜 2명을 찾아내기 위한 이중 반복문
fake1 = None
fake2 = None

### 첫 번째 가짜 후보 i
for i in range(9):
### 두 번째 가짜 후보 j (i 다음 사람부터 확인하면 중복이 없어요)
    for j in range(i + 1, 9):
        # 전체 합에서 두 명의 키를 뺐을 때 100이 되는지 확인
        if total_sum - nine_dwarfs[i] - nine_dwarfs[j] == 100:
            fake1 = nine_dwarfs[i]
            fake2 = nine_dwarfs[j]
            break # 안쪽 루프 탈출
    if fake1 is not None:
        break # 바깥쪽 루프 탈출

### 4. 가짜 2명을 리스트에서 제거
nine_dwarfs.remove(fake1)
nine_dwarfs.remove(fake2)

### 5. 남은 일곱 난쟁이의 키를 오름차순으로 정렬해서 출력
nine_dwarfs.sort()

for height in nine_dwarfs:
    print(height)
